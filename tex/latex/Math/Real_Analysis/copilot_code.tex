% \usepackage{transparent}

% Define colors
\definecolor{horange}{RGB}{245,92,0} % Define your desired orange color
\definecolor{CTV}{RGB}{255,255,255} % Color for title text
\definecolor{CNoV}{RGB}{255,255,255} % Color for chapter number

% Define a custom Mandelbrot-like shading
\pgfdeclarehorizontalshading{mandelbrotshading}{100bp}{
    color(0bp)=(orange!70!red);
    color(25bp)=(red!70!gray);
    color(50bp)=(gray!50!black);
    color(75bp)=(black!80!gray);
    color(100bp)=(black)
}

% Define a fading based on the custom shading
\tikzfading[name=mandelbrotfading, left color=transparent!0, right color=transparent!100]

% Define the gradient command
\newcommand{\gradient}[1]{
    \begin{tikzpicture}
        % Draw a rectangle with the custom shading and fading
        \shade[shading=mandelbrotshading]
            (0,0) rectangle (\linewidth,2.5cm);
        
        % Add the chapter title text
        \node[anchor=south west, font=\CTV, text=white, align=left]
            at (0.5cm,1.5cm) {\textcolor{horange}{#1}};
        
        % Add the chapter number if it exists
        \ifnum \thechapter>0
            \node[anchor=center, font=\CNoV, text=white]
                at (\linewidth-2cm,1.5cm) {\textcolor{horange}{\thechapter}};
        \fi
    \end{tikzpicture}%
    \vskip 40pt
}

% Redefine chapter commands to use the gradient
\renewcommand{\DOCH}{} % Clear the default chapter command
\renewcommand{\DOTI}[1]{\gradient{#1}} % Redefine for numbered chapters
\renewcommand{\DOTIS}[1]{\gradient{#1}} % Redefine for unnumbered chapters

% Adjust chapter spacing
\makeatletter
\patchcmd{\@makechapterhead}{\vspace*{50\p@}}{\vspace*{-20\p@}}{}{}
\patchcmd{\@makeschapterhead}{\vspace*{50\p@}}{\vspace*{-20\p@}}{}{}
\patchcmd{\DOTI}{\vskip 80\p@}{\vskip 40\p@}{}{}
\makeatother
