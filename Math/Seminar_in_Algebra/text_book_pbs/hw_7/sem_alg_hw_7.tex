\section*{Section 29}

\textbf{Computations}

In Exercise 1, show that the given number \(\alpha \in \C\) is algebraic over \(\Q\) by finding \(f(x) \in \Q[x]\) such that \(f(\alpha) = 0\).  

\begin{enumerate}
	\item 1 + \(\sqrt{2}\)
	\sol{
		% Following what we did in class, to find an \(f(x)\) such that \(f(\alpha) = 0\), we start by squaring \(\alpha\): \(\alpha^{2} = 3 + 2\sqrt{2}\). Then, we try \(\alpha^{2} -3\) to get \(2\sqrt{2}\) alone. Then, we know \((\alpha^{2} -3)^{2}\) is just \((2\sqrt{2})^{2} = 8\). Expanding \((\alpha^{2} -3)^{2}\), we find 
		% \[
		% 	(1 + \sqrt{2})^{4} - 6(1 - \sqrt{2})^{2} + 9 = 8.
		% \]
		% Thus, we get the polynomial \(\alpha^{4} - 6\alpha^{2} + 1\).
		Set \(\alpha = 1 + \sqrt{2}\) and solve \(\alpha = 0\) by computing the following:
		\begin{align*}
			\alpha &= 1 + \sqrt{2} \\
			\alpha - 1 &= \sqrt{2} \\
			\alpha^{2} - 2\alpha - 1 &= 0.
		\end{align*}
		Thus, we have shown \(f(\alpha) = 0\) with \(x^{2} - 2x - 1 \in \Q[x]\).
	}
\end{enumerate}

In Exercise 6, find \(\irr{\alpha, \Q}\) and \(\deg(\alpha, \Q)\) for the given algebraic number \(\alpha \in \C\). Be prepared to prove that your polynomials are irreducible over \(\Q\) if challenged to do so.

\begin{enumerate}
	\setcounter{enumi}{5}
	\item \(\sqrt{3 - \sqrt{6}}\)
	\sol{
		To find the irreducible polynomial, we set \(\alpha = \sqrt{3 - \sqrt{6}}\) and solve:
		\begin{align*}
			\alpha^{2} - 3 &= - \sqrt{6} \\
			\alpha^{4} - 6 \alpha^{2} + 3 &= 0.
		\end{align*}
		This polynomial has degree 4, and is irreducible by Theorem 23.15 for \(p = 3\). 
	}
\end{enumerate}

\textbf{Theory}

\begin{enumerate}
	\setcounter{enumi}{29}
	\item Let \(E\) be an extension field of a finite field \(F\), where \(F\) has \(q\) elements. Let \(\alpha \in E\) be algebraic over \(F\) of degree \(n\). Prove that \(F(\alpha)\) has \(q^{n}\) elements. 
	\pf{
		From Theorem 29.18, every element in \(F(\alpha)\) can be expressed uniquely as \(\beta = b_{0} + b_{1} \alpha + \cdots + b_{n - 1}\alpha^{n - 1},\) for \(b_{i} \in F\). Since there are \(q\) elements in \(F\), there are \(q\) choices for each \(b_{i}\). Because there are \(n\) such \(b_{i}\) (\(b_{0}\) to \(b_{n - 1}\)), there are \(q \times q \times \cdots q \ (n \text{ times})\) total combinations. This gives \(q^{n}\) elements in \(F(\alpha)\).
