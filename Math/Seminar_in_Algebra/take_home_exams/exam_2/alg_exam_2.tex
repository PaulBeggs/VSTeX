\begin{enumerate}
	\item Recall that the polynomial \(x^{2} + 1\) is irreducible in \(\Z_{3}[x]\), so \(\Z_{3}[x] / \gen{x^{2} + 1}\) is a field. Consider the isomorphic field extension \(\Z_{3}(\alpha)\) of \(\Z_{3}\), where \(\alpha\) is the root of \(x^{2} + 1\).
	\begin{enumerate}
		\item List all elements of \(\Z_{3}(\alpha)\)
		\sol{
			By Exercise 30 in Section 29, we know there are \(3^{2} = 9\) elements in the field \(\Z_{3}(\alpha)\). Then, by Theorem 29.18, we know they can be uniquely expressed as 
			\[
				b_{0} + b_{1}\alpha, 
			\]
			for \(b_{i} \in \Z_{3}\). Thus, we get a list of elements:
			\[
				\{0,\ 1,\ 2,\ 0 + \alpha,\ 1 + \alpha,\ 2 + \alpha,\ 2\alpha,\ 1 + 2\alpha,\ 2 + 2\alpha\} 
			\]
		}

		\item Compute \((a + b\alpha)(c + d\alpha)\) for arbitrary \(a,b,c,d \in \Z_{3}\), simplifying your answer using the fact that \(\alpha^{2} = -1\). 
		\sol{
			We compute
			\begin{align*}
				(a + b\alpha)(c + d\alpha) &= ac + ad\alpha + bc\alpha + bd\alpha^{2} \\
				&= ac + \alpha(ad + bc) - bd
			\end{align*}
		}

		\item Using your result from part (b), construct the addition and multiplication tables for \(\Z_{3}(\alpha)\). 
		\sol{
			The addition and multiplication tables are presented in \hyperref[tab:add]{Table 1} and \hyperref[tab:multi]{Table 2} respectively.
		}

	\end{enumerate}
	
	\item Let \(p\) be a prime number and let \(\alpha \in \R\) be a root of the polynomial \(x^{p} - 2\). Set \(F = \Q(\alpha)\).
	\begin{enumerate}
		\item Show that \([F : \Q] = p\).
		\pf{
			Let \(x^{p} - 2 = g(x) \in \Q[x]\) with \(\alpha \in \R\) a root. This polynomial is irreducible over \(\Q\) by Eisenstein's criterion for prime \(q = 2\). Because \(g(x)\) is a monic, irreducible polynomial with \(g(\alpha) = 0\), it is the minimal polynomial for \(\alpha\) over \(\Q\), which means \(\deg(\alpha, \Q) = p\). Furthermore, since \(\R\) is an extension field of \(\Q\) and \(\alpha\) is algebraic over \(\Q\), then by Theorem 30.23, \(F = \Q(\alpha)\) is a \(p\)-dimensional vector space over \(\Q\). Thus, \([F : \Q] = p\).
		}
\newpage
		\item Show that the only subfields of \(F\) are \(\Q\) and \(F\). (\textit{Hint: Let \(F'\) be a subfield of \(F\) and create a tower of fields with \(F'\).})
		\pf{
			Suppose that \(F'\) is a subfield of \(F\). This creates a tower of fields \(\Q \subseteq F' \subseteq F\). Then, by Theorem 31.4, we have that 
			\begin{equation}
				[F : \Q] = [F : F'][F' : \Q].
			\end{equation}
			We know from part (a) that \([F : \Q] = p\), and since \(p\) is prime, it only factors into \(1 \cdot p\). Thus, we have two cases:
			\begin{itemize}
				\item \textbf{Case 1:} If \([F' : \Q] = 1\), then \(F' = \Q\).
				\item \textbf{Case 2:} If \([F' : \Q] = p\), then \([F : F'] = 1\) because \([F : \Q] = p\). Thus, \(F' = F\).
			\end{itemize}
			Therefore, in either case, \(F'\) is either \(F\) or \(\Q\). Hence, the only subfields of \(F\) are \(\Q\) and \(F\).
		}

		\item Let \(\beta \in F\ \backslash\ \Q\). Show that \(\Q(\beta) = F\), and conclude that \(\alpha = p(\beta)\) for some polynomial \(p(x) \in \Q[x]\).
		\pf{
			Suppose \(\beta \in F\ \backslash\ \Q\). From part (a), we know that \(F\) is a simple extension of \(\Q\) with \(\alpha\) algebraic over \(\Q\) with \(\deg(\alpha, \Q) = p\). Then, from Theorem 29.18, we know every element in \(F\) can be expressed uniquely as 
			\[
				b_{0} + b_{1}\alpha + \cdots + b_{p - 1}\alpha^{p-1},
			\]
			for \(b_{i} \in \Q\). Now, since \(\beta \in F\ \backslash\ \Q\), we know \(\beta \notin \Q\). Therefore, \(\beta\) cannot be the constant term \(b_{0}\), meaning there must be at least one coefficient \(b_{1}\) through \(b_{p - 1}\) that is nonzero. Then, Theorem 30.23 tells us that this \(\beta \in F\) is algebraic over \(\Q\), and \(\deg(\beta, \Q) \le \deg(\alpha, \Q)\). Thus, we can create a tower of fields \(\Q \subseteq \Q(\beta) \subseteq F\). Then, part (b) tells us that \(\Q(\beta)\) must be equal to either \(\Q\) or \(F\). Because \(\beta \notin \Q\), the field \(\Q(\beta)\) consists of elements that are outside \(\Q\), so \(\Q(\beta) \ne \Q\). Thus, it must be that \(\Q(\beta) = F\).

			Since \(\Q(\beta) = F\), it follows that \(\alpha \in \Q(\beta)\). So, by Theorem 29.18, we can express this \(\alpha\) as a unique polynomial \(p(\beta)\) for \(p(x) \in \Q[x]\). \(\mathbb{F}\)
		}
		
	\end{enumerate} 
\end{enumerate}
