\begin{enumerate}
	\item In your own words, prove Theorem 19.15 on pg. 182.
	\pf{
		Let \(R\) be a ring with unity. We will begin by proving the first statement. Assume \(n \cdot 1 \ne 0\) for all \(n \in \Z^{+}\). Suppose, for the sake of contradiction, that \(R\) has a positive characteristic \(k\) such that \(k \cdot a = 0\) for every \(a \in R\). Since \(R\) is a ring with unity, \(1 \in R\). Therefore, it must be the case that \(k \cdot 1 = 0\). This leads to a contradiction, because we said that for all positive integers \(n \cdot 1 \ne 0\). Therefore, no positive \(k\) can exist, and by the new definition, \(R\) has characteristic 0. 

		Now, for the second statement. Assume \(n \cdot 1 = 0\) for all \(n \in \Z^{+}\). Then, for any \(a \in R\), consider the computation:
		\begin{align*}
			n \cdot a &= \underbrace{a + a + a + \cdots}_{n \text{ times}} & \text{definition of } n \cdot a, \\
			&= a(\underbrace{1 + 1 + \cdots}_{n \text{ times}}) & \text{left distributive law}, \\
			&= a(n \cdot 1) \\
			&= a(0) & \text{by assumption } n \cdot 1 = 0, \\
			&= 0 & \text{by Theorem 18.8}.
		\end{align*}
		Thus, we have shown for all \(a \in R\), that \(n \cdot a = 0\), so \(R\) is of characteristic \(n\).  
	}
	\item Find the characteristic of the following rings:
	\begin{enumerate}
		\item \(\Z_{4} \times 4\Z\)
		\sol{
			We know this ring has characteristic of 0 because we cannot find an \(n\) such that \(n \cdot (x,y) = (0,0)\) for all \(x \in \Z_{4}\) and \(y \in 4\Z\). For a counterexample, consider \(y = 4\). Since \(4\Z\) is infinite, for any positive integer \(n\), \(n \cdot 4\) will result in a multiple greater than or equal to \(4\). Thus, there does not exist an \(n\) such that \(n \cdot (x,4) = (0,0)\), so \(\Z_{4} \times 4\Z\) is of characteristic 0.
		}
		\item \(\Z_{2} \times \Z_{6}\)
		\sol{
			This ring is of characteristic 6 by the new definition. To show this, let \(x \in \Z_{2}\) and \(y \in \Z_{6}\). Then, consider the following:
			\[
				6 \cdot (x,y) = (6x \bmod{2}, 6y \bmod{6}) = (0, 0).
			\]
		}
	\end{enumerate} 
	\item Find an example of an infinite order ring that is not of characteristic 0.
	\sol{
		The polynomial ring \(\Z_{3}[x]\) is an infinite order ring that of characteristic 3. First note that \(\Z_{3}[x]\) is of infinite order because it consists of an infinite number of polynomials. Then, \(\Z_{3}[x]\) has unity, so we can use Theorem 19.15. Thus, we can use the 0 degree polynomial 1, which is in \(\Z_{3}[x]\). We find that for the equation \(n \cdot 1 = 0\), \(n = 3\) is the smallest positive integer that works. 
	}
\end{enumerate}

\newpage

\begin{enumerate}
	\setcounter{enumi}{3}
	\item We will show that every ring \(R\) can be enlarged to a ring \(S\) with unity, having the same characteristic as \(R\). If \(R\) has characteristic 0, set \(S = R \times \Z\); otherwise (if \(R\) has characteristic \(n\)), set \(S = R \times \Z_{n}\). Let addition in \(S\) be the same usual addition by components, and let multiplication be defined by
	\[
		(r_{1}, n_{1})(r_{2}, n_{2}) = (r_{1}r_{2} + n_{1} \cdot r_{2} + n_{2} \cdot r_{1}, n_{1}n_{2}),
	\]
	where \(n \cdot r\) is defined above.
	\begin{enumerate}
		\item Show the left distributive law for \(S\). (One can then go on to prove that \(S\) is indeed a ring.)
		\sol{
			To show the left distributive law for \(S\), we need to show that for \(a,b,c \in R\) and \(x,y,z \in \Z\), that 
			\[
				(a,x)[(b,y) + (c,z)] = (a,x)(b,y) + (a,x)(c,z).
			\]
			Thus, consider the following:
			\begin{align}
				(a,x)[(b,y) + (c,z)] &= (a,x)(b + c, y + z) \\
				&= (a(b + c) + x \cdot (b + c) + (y + z) \cdot a, x(y + z)) \\
				&= (ab + ac + x \cdot b + x \cdot c + y \cdot a + z \cdot a, xy + xz) \\
				&= (ab + x \cdot b + y \cdot a, xy) + (ac + x \cdot c + z\cdot a, xz) \\
				&= (a,x)(b,y) + (a,x)(c,z).
			\end{align}
			Note that (1) and (4) are from component-wise addition; (2) and (5) are by definition of multiplication; and (3) includes the distributive law for \(R\) with \(a(b + c) = ab + ac\), the left distributive law for integers with \(x(y + z) = xy + xz\), and definition of \(n \cdot r\) from above with \(x \cdot (b + c) = x \cdot b + x \cdot c\) and \((y + z) \cdot a = y \cdot a + z \cdot a\).

			Therefore, we have shown that the left distribution laws hold for \(S\). (And we assume that \(S\) is a ring.)
		}
\newpage
		\item Show that \(S\) has unity.
		\sol{
			To show that \(S\) has unity, we will show \((0,1)(a,x) = (a,x)(0,1) = (a,x)\). We choose \((0,1)\) as the unity because the only element that we know is in \(R\) is the additive identity, 0. We pick \(1 \in \Z\) (or \(\Z_{n}\)) because that is the unity in that ring. Thus, consider the computations
			\begin{align*}
				(0,1)(a,x) &= (0a + 1 \cdot a + x \cdot 0, (1)x) \\
				&= (a, x),
			\end{align*}
			and
			\begin{align*}
				(a,x)(0,1) &= (a0 + x \cdot 0 + 1 \cdot a, (x)1) \\
				&= (a,x).
			\end{align*}
			Thus, \(S\) has unity \((0,1)\).
		}
		\item Show that \(S\) and \(R\) have the same characteristic.
		\sol{
			Since \(S\) is a ring with unity, we can use Theorem 19.15 to determine \(S\)'s characteristic. Consider the equation \(n \cdot (0,1)\). We have two cases:
			\begin{itemize}
				\item \textbf{\(R\) has characteristic 0:} Thus, \(S = R \times \Z\). Hence, for any \(n \in \Z^{+}\), \(n \cdot (0,1) = (0,n) \ne (0,0)\). Thus, \(S\) has characteristic 0.
				\item \textbf{\(R\) has characteristic \(n\):} Thus, \(S = R \times \Z_{n}\). Let \(n\) be equal to the modulus in \(\Z_{n}\). Then, \(n \cdot (0,1) = (0,n \mod n) = (0,0)\). Thus, \(S\) has characteristic \(n\).
			\end{itemize}
			Therefore, in either case, \(S\) and \(R\) have the same characteristic. 
		}
	\end{enumerate}
\end{enumerate}
