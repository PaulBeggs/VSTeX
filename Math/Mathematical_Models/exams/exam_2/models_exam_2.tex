\section*{Chapter 8}


\begin{enumerate}
    \setcounter{enumi}{13}
    \item (Geometric distribution) In this problem, we investigate the discrete version of the exponential distribution. Suppose that arrivals occur at random times \(i = 1,2,3,\cdots\). Then, the time \(X\) between two successive arrivals has the geometric distribution
    \[
        P(X = i) = p(1 - p)^{i - 1},
    \]
    where \(p\) is the probability of an arrival occurring at time \(i\).
    \begin{enumerate}
        \item Show that \(P(X > i) = (1 - p)^{i}\). [Hint: Use the geometric series \(1 + x + x^{2} + x^{3} + \cdots = (1 - x)^{-1}\).]
        \sol{
            Let \(q = (1 - p)\) so that \(P(X = i) = p(1 - p)^{i - 1} = pq^{i - 1}\). The probability \(P(X > i)\) is the sum of probabilities that \(X\) takes any value from \(i + 1\) to \(\infty\). Thus, we have:
            \[
                P(X > i) = \sum_{k = i + 1}^{\infty} pq^{k - 1}.
            \]
            Since \(p\) does not depend on \(k\), we can move it outside the sum:
            \[
                P(X > i) = p \sum_{k = i + 1}^{\infty} q^{k - 1} = q^{(i + 1) - 1} + q^{(i + 2) - 1} + \cdots.
            \]
            Factoring out the \(q^{i}\) term, we get:
            \[
                P(X > i) = pq^{i} \sum_{k = i + 1}^{\infty} = 1 + q + q^{2} + \cdots.
            \]
            Using the hint, we are left with \(pq^{i}(1 - q)^{-1}\). Substituting \((1 - p)\) back for \(q\), and we find:
            \[
                P(X > i) = pq^{i}(1 - q)^{-1} = p\frac{(1 - p)^{i}}{(1 - (1 - p))} = (1 - p)^{i}.
            \]
            
        }
        \item Use part (a) to show that \(X\) has the memoryless property \(P(X > i + j \mid X > j) = P(X > i)\).
        \sol{
            Consider the following:
            \begin{align*}
                P(X > i + j \mid X > j) &= \frac{P(X > i + j \text{ and } X > i)}{P(X > i)} \\
                &= \frac{P(X > i + j)}{P(X > j)} \\
                &= \frac{q^{i + j}}{q^{j}} \\
                &= q^{i} \\
                &= P(X > i).
            \end{align*}
        }
        \item Compute \(E[X] = 1/p\). [Hint: Differentiate the geometric series to obtain \(1 + 2x + 3x^{2} + \cdots = (1 - x)^{-2}\).] Explain why \(p\) is the arrival rate for the discrete process.
        \sol{
            Recall for a discrete distribution that 
            \[
                E[X] = \sum_{i = 1}^{\infty} iP(X = i),
            \]
            so we get 
            \[
                E[X] = \sum_{i = 1}^{\infty} i pq^{i - 1}.
            \]
            Again, since \(p\) does not depend on \(i\), we can factor it out:
            \[
                E[X] = p\sum_{i = 1}^{\infty} iq^{i - 1} = 1 + 2q^{1} + 3q^{2} + \cdots.
            \]
        }
    \end{enumerate}
\end{enumerate}
